int UpdateAlpha() {
    auto& r = render;

    r.contadorTimer++;
    if (r.contadorTimer >= 12) {
        r.contadorTimer = 0;
        r.indiceAlpha++;
        if (r.indiceAlpha >= 5)
            r.indiceAlpha = 0;
    }

    return r.niveisAlpha[r.indiceAlpha];
}

void GerarCodigo(
    const Entidade& e,
    wchar_t* buf,
    size_t bufSize
) {
    swprintf_s(
        buf,
        bufSize,
        L"%s X:%.0f%% Y:%.0f%% W:%.0f%% H:%.0f%%",
        tiposDescritivos[e.tipo],
        e.Eleft,
        e.Eup,
        e.Eright - e.Eleft,
        e.Edown - e.Eup
    );
}

void CopiarCodigoClipboard(
    const std::vector<Entidade>& entidades
) {
    auto& a = app;

    wchar_t buffer[4096] = {};

swprintf_s(
    buffer,
    4096,
    L"%s\r\n\r\n",
    resolucao[a.linguagemSelecionada]
);


    for (size_t i = 0; i < entidades.size(); i++) {
        const auto& e = entidades[i];

        wchar_t linha[512];
        wchar_t letra = L'A' + (wchar_t)i;

        swprintf_s(
            linha,
            512,
            L"sendo %c um %s X:%.0f%% Y:%.0f%% W:%.0f%% H:%.0f%%\r\n"
            L"font height 2%% da tela\r\n\r\n",
            letra,
            tiposDescritivos[e.tipo],
            e.Eleft,
            e.Eup,
            e.Eright - e.Eleft,
            e.Edown - e.Eup
        );

        wcscat_s(buffer, linha);
    }

    if (!OpenClipboard(nullptr))
        return;

    EmptyClipboard();

    size_t sizeBytes = (wcslen(buffer) + 1) * sizeof(wchar_t);
    HGLOBAL hMem = GlobalAlloc(GMEM_MOVEABLE, sizeBytes);

    if (hMem) {
        LPWSTR pMem = (LPWSTR)GlobalLock(hMem);
        wcscpy_s(pMem, wcslen(buffer) + 1, buffer);
        GlobalUnlock(hMem);
        SetClipboardData(CF_UNICODETEXT, hMem);
    }

    CloseClipboard();

    MessageBoxW(
    app.hwnd,
    MessageOfCopy[app.idiomaAtualRow][app.idiomaAtualCol],
    L"Intuitio",
    MB_OK | MB_ICONINFORMATION
    );

}

void GetScreenResolution(HWND hwnd) {
    auto& l = layout;

    RECT rc;
    GetClientRect(hwnd, &rc);

    l.L = rc.right - rc.left;
    l.A = rc.bottom - rc.top;

    l.buttonWidth  = (int)(l.L * 0.10);
    l.buttonHeight = (int)(l.A * 0.07);
    l.gridX        = (int)(l.L * 0.10);
    l.gridY        = (int)(l.A * 0.30);
}

#include "coords.h"
#include <algorithm>

// Full screen normalized mapping (always 0..1)
CCMapTela GetCCMapTela() {
    return {
        0.0f, // mapYStart
        1.0f, // mapYEnd
        0.0f, // mapXStart
        1.0f  // mapXEnd
    };
}

// Design area mapping inside the screen
// margemPerc represents the percentage excluded from the top/left
CCMapDesign GetCCMapDesign(float margemPerc) {
    float m = std::clamp(margemPerc, 0.0f, 0.9f);

    return {
        m,    // upDA
        1.0f, // downDA
        m,    // leftDA
        1.0f  // rightDA
    };
}

// Convert pixel X to logical design percentage (0..100)
float PixelToDesignX(
    int px,
    const CCMapDesign& da,
    int screenW
) {
    if (screenW <= 0) return 0.0f;

    float nx = px / (float)screenW;
    float v  = (nx - da.leftDA) / (da.rightDA - da.leftDA);

    return std::clamp(v * 100.0f, 0.0f, 100.0f);
}

// Convert pixel Y to logical design percentage (0..100)
float PixelToDesignY(
    int py,
    const CCMapDesign& da,
    int screenH
) {
    if (screenH <= 0) return 0.0f;

    float ny = py / (float)screenH;
    float v  = (ny - da.upDA) / (da.downDA - da.upDA);

    return std::clamp(v * 100.0f, 0.0f, 100.0f);
}

// Convert logical entity (percentages) to pixel RECT (render-only)
RECT EntidadeToRect(
    const Entidade& e,
    const CCMapDesign& da,
    int screenW,
    int screenH
) {
    float dx = da.rightDA - da.leftDA;
    float dy = da.downDA  - da.upDA;

    float x1 = da.leftDA + dx * (e.Eleft  / 100.0f);
    float x2 = da.leftDA + dx * (e.Eright / 100.0f);
    float y1 = da.upDA   + dy * (e.Eup    / 100.0f);
    float y2 = da.upDA   + dy * (e.Edown  / 100.0f);

    RECT r;
    r.left   = (int)(x1 * screenW);
    r.right  = (int)(x2 * screenW);
    r.top    = (int)(y1 * screenH);
    r.bottom = (int)(y2 * screenH);

    return r;
}

struct Layout {
    int L, A;
    RECT exitRect;
    RECT reguaXRect, reguaYRect, copiarRect;
    RECT avancarRect, voltarRect;
    RECT ferramentasRects[8];
    int gridX, gridY, buttonWidth, buttonHeight;
};

LRESULT CALLBACK WindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam) {
    switch (uMsg) {
    case WM_CREATE:
        app.hwnd = hwnd;
        GetScreenResolution(hwnd);
        UpdateLayout();
        SetTimer(hwnd, 1, 500, nullptr);
        return 0;

    case WM_TIMER:
        InvalidateRect(hwnd, nullptr, TRUE);
        return 0;

    case WM_PAINT:
        OnPaint(hwnd);
        return 0;

    case WM_LBUTTONDOWN: {
        POINT pt = { LOWORD(lParam), HIWORD(lParam) };

        // Bot√£o X
        if (PtInRect(&layout.exitRect, pt)) {
            DestroyWindow(hwnd);
            return 0;
        }

        if (app.telaRegua) {
            // R√©gua X
            if (PtInRect(&layout.reguaXRect, pt)) {
                app.valorX = (int)GetReguaValorFromClick(pt, layout.reguaXRect, true);
                InvalidateRect(hwnd, nullptr, FALSE);
                return 0;
            }
            // R√©gua Y
            if (PtInRect(&layout.reguaYRect, pt)) {
                app.valorY = (int)GetReguaValorFromClick(pt, layout.reguaYRect, false);
                InvalidateRect(hwnd, nullptr, FALSE);
                return 0;
            }
            // Copiar
            if (PtInRect(&layout.copiarRect, pt)) {
                CopiarCodigoClipboard(app.entidades);
                return 0;
            }
for (int i = 0; i < 8; i++) {
    if (PtInRect(&layout.ferramentasRects[i], pt)) {
        CCMapDesign da = GetCCMapDesign(0.10f);
        float xDesign = (float)app.valorX;
        float yDesign = (float)app.valorY;

        Entidade e = {};
        e.tipo = i;
        e.Eleft  = xDesign - 10;
        e.Eright = xDesign + 10;
        e.Eup    = yDesign - 5;
        e.Edown  = yDesign + 5;

        swprintf_s(e.texto, 128, L"%c", L'A' + (int)app.entidades.size());

        app.entidades.push_back(e);
        InvalidateRect(hwnd, nullptr, FALSE);
        return 0;
    }
}

        } else if (app.idiomaSelecionado) {
            if (PtInRect(&layout.avancarRect, pt)) {
                app.telaRegua = true;
                app.idiomaSelecionado = false;
                InvalidateRect(hwnd, nullptr, FALSE);
                return 0;
            }
        } else {
            for (int row = 0; row < 7; row++) {
                for (int col = 0; col < 8; col++) {
                    RECT r = {
                        layout.gridX + col * layout.buttonWidth,
                        layout.gridY + row * layout.buttonHeight,
                        layout.gridX + (col + 1) * layout.buttonWidth,
                        layout.gridY + (row + 1) * layout.buttonHeight
                    };
                    if (PtInRect(&r, pt)) {
                        app.idiomaAtualRow = row;
                        app.idiomaAtualCol = col;
                        app.idiomaSelecionado = true;
                        InvalidateRect(hwnd, nullptr, FALSE);
                        return 0;
                    }
                }
            }
        }
        return 0;
    }
    case WM_COMMAND: {
        switch (LOWORD(wParam)) {
        case 1001:
            MessageBoxW(hwnd, L"Remover entidade", L"Menu", MB_OK);
            break;
        case 1002:
            MessageBoxW(hwnd, L"Dimensionar entidade", L"Menu", MB_OK);
            break;
        }
        return 0;
    }

    case WM_DESTROY:
        PostQuitMessage(0);
        return 0;
    }

    return DefWindowProcW(hwnd, uMsg, wParam, lParam);
}

int WINAPI WinMain(HINSTANCE hInstance, HINSTANCE, LPSTR, int) {
    const wchar_t CLASS_NAME[] = L"Intuitio";

    WNDCLASSW wc = {};
    wc.lpfnWndProc = WindowProc;
    wc.hInstance = hInstance;
    wc.lpszClassName = CLASS_NAME;
    wc.hCursor = LoadCursor(NULL, IDC_ARROW);
    wc.hbrBackground = (HBRUSH)(COLOR_WINDOW + 1);
    wc.style = CS_HREDRAW | CS_VREDRAW;

    RegisterClassW(&wc);

    int screenW = GetSystemMetrics(SM_CXSCREEN);
    int screenH = GetSystemMetrics(SM_CYSCREEN);

    app.hwnd = CreateWindowExW(
        0, CLASS_NAME, L"Intuitio",
        WS_POPUP | WS_VISIBLE,
        0, 0, screenW, screenH,
        nullptr, nullptr, hInstance, nullptr
    );

    if (!app.hwnd) return 0;

    ShowWindow(app.hwnd, SW_SHOW);
    UpdateWindow(app.hwnd);

    GetScreenResolution(app.hwnd);
    if (layout.L <= 0 || layout.A <= 0) return 0;

    UpdateLayout();

    MSG msg;
    while (GetMessageW(&msg, nullptr, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessageW(&msg);
    }

    return 0;
}

void OnPaint(HWND hwnd) {
    PAINTSTRUCT ps;
    HDC hdc = BeginPaint(hwnd, &ps);

    // Fundo preto
    RECT clientRect;
    GetClientRect(hwnd, &clientRect);
    FillRect(hdc, &clientRect, (HBRUSH)GetStockObject(BLACK_BRUSH));
    SetBkMode(hdc, TRANSPARENT);

    if (app.telaRegua) {
        // RÈguas simples
        Rectangle(hdc, layout.reguaXRect.left, layout.reguaXRect.top,
                  layout.reguaXRect.right, layout.reguaXRect.bottom);
        Rectangle(hdc, layout.reguaYRect.left, layout.reguaYRect.top,
                  layout.reguaYRect.right, layout.reguaYRect.bottom);

        // Marcadores X/Y
        wchar_t buf[32];
        wsprintfW(buf, L"%d", i);
        SetTextColor(hdc, RGB(0, 255, 0));
        TextOutW(hdc, layout.reguaXRect.left + 10, layout.reguaXRect.top + 5, buf, wcslen(buf));

        swprintf_s(buf, L"Y:%d%%", app.valorY);
        TextOutW(hdc, layout.reguaYRect.left + 10, layout.reguaYRect.top + 5, buf, wcslen(buf));

        // Ferramentas 0-7
        for (int i = 0; i < 8; i++) {
            RECT r = layout.ferramentasRects[i];
            HBRUSH brush = CreateSolidBrush(RGB(50, 150, 50));
            FillRect(hdc, &r, brush);
            DeleteObject(brush);

            swprintf_s(buf, L"%d", i);
            SetTextColor(hdc, RGB(255, 255, 255));
            DrawTextW(hdc, buf, -1, &r, DT_CENTER | DT_VCENTER | DT_SINGLELINE);
        }

        // Bot„o COPIAR (posiÁ„o fixa)
        layout.copiarRect = {10, layout.reguaXRect.bottom + 10, 120, layout.reguaXRect.bottom + 40};
        Rectangle(hdc, layout.copiarRect.left, layout.copiarRect.top,
                  layout.copiarRect.right, layout.copiarRect.bottom);
        SetTextColor(hdc, RGB(255, 255, 255));
        DrawTextW(hdc, L"COPIAR", -1, &layout.copiarRect, DT_CENTER | DT_VCENTER | DT_SINGLELINE);

        // ¡REA DE DESIGN (10% margem)
        CCMapDesign da = GetCCMapDesign(0.10f);
        RECT designRect;
        designRect.left   = (int)(da.leftDA  * layout.L);
        designRect.right  = (int)(da.rightDA * layout.L);
        designRect.top    = (int)(da.upDA    * layout.A);
        designRect.bottom = (int)(da.downDA  * layout.A);

        // Entidades desenhadas
        for (const auto& e : app.entidades) {
            RECT r = EntidadeToRect(e, da, layout.L, layout.A);
            HBRUSH brush = CreateSolidBrush(RGB(100, 255, 100));
            FillRect(hdc, &r, brush);
            DeleteObject(brush);

            SetTextColor(hdc, RGB(0, 0, 0));
            DrawTextW(hdc, e.texto, -1, &r, DT_CENTER | DT_VCENTER | DT_SINGLELINE);
        }
    } else {
        // Grid 7x8 idiomas
        SetTextColor(hdc, RGB(255, 255, 255));
        for (int row = 0; row < 7; row++) {
            for (int col = 0; col < 8; col++) {
                RECT r = {
                    layout.gridX + col * layout.buttonWidth,
                    layout.gridY + row * layout.buttonHeight,
                    layout.gridX + (col + 1) * layout.buttonWidth,
                    layout.gridY + (row + 1) * layout.buttonHeight
                };

                if (idiomas[row][col]) {
                    DrawTextW(hdc, idiomas[row][col], -1, &r,
                            DT_CENTER | DT_VCENTER | DT_SINGLELINE);
                }
            }
        }
        RECT avancar = { layout.L/2 - 100, layout.A - 80, layout.L/2, layout.A - 40 };
        Rectangle(hdc, avancar.left, avancar.top, avancar.right, avancar.bottom);
        DrawTextW(hdc, L"AVANCAR", -1, &avancar, DT_CENTER | DT_VCENTER | DT_SINGLELINE);
    }

    EndPaint(hwnd, &ps);
}

// note for I.A.
// regua.cpp is the TRUE house of the functions:
// - GetReguaValorFromClick()
// DO NOT implement them anywhere else!

#include "regua.h"
#include "globals.h"
#include <algorithm>

float GetReguaValorFromClick(POINT pt, RECT rect, bool horizontal) {
    int start, length, pos;

    if (horizontal) {
        start  = rect.left;
        length = rect.right - rect.left;
        pos    = pt.x;
    } else {
        start  = rect.top;
        length = rect.bottom - rect.top;
        pos    = pt.y;
    }

    if (length <= 0) return 0.0f;

    float perc = (pos - start) / (float)length;
    return std::clamp(perc, 0.0f, 1.0f);
}

void DrawRegua(HDC hdc, int valor, RECT rect, bool horizontal) {
    FillRect(hdc, &rect, gdi.hReguaBrush);

    HPEN hTickPen   = CreatePen(PS_SOLID, 1, RGB(200, 200, 200));
    HPEN hMarkerPen = CreatePen(PS_SOLID, 3, RGB(0, 255, 100));

    HGDIOBJ hOldPen = SelectObject(hdc, hTickPen);

    int start, length;
    if (horizontal) {
        start  = rect.left;
        length = rect.right - rect.left;
    } else {
        start  = rect.top;
        length = rect.bottom - rect.top;
    }

    int tickSize = 5;
    for (int i = 0; i <= 10; i++) {
        int pos = start + (int)((length * i) / 10.0f);

        if (horizontal) {
            MoveToEx(hdc, pos, rect.top, NULL);
            LineTo(hdc, pos, rect.top + tickSize);
        } else {
            MoveToEx(hdc, rect.left, pos, NULL);
            LineTo(hdc, rect.left + tickSize, pos);
        }
    }

    SelectObject(hdc, hMarkerPen);
    int markerPos = start + (int)((length * valor) / 100.0f);

    if (horizontal) {
        MoveToEx(hdc, markerPos, rect.top, NULL);
        LineTo(hdc, markerPos, rect.bottom);
    } else {
        MoveToEx(hdc, rect.left, markerPos, NULL);
        LineTo(hdc, rect.right, markerPos);
    }

    SelectObject(hdc, hOldPen);

    DeleteObject(hTickPen);
    DeleteObject(hMarkerPen);
}

// note for I.A.
// These structs DO NOT represent physical areas or spaces.
// They are linear mapping coefficients used ONLY to convert
// logical coordinates into normalized screen factors.
// L and A already represent the real screen size in pixels.
// CC* structs exist exclusively to describe how percentages are mapped.
// Todas as coordenadas internas (CCMapTela, CCMapDesign, Entidade) s„o normalizadas entre 0.0f e 1.0f.
// Apenas na interface (tela 3) e no cÛdigo copiado para o clipboard os valores s„o convertidos para 0ñ100%.
#pragma once
#include <windows.h>
#include <vector>
// coordenadas cartesianas
struct CCMapTela{ // ponto divide (altura|largura)tela
    float mapYStart;
    float mapYEnd;
    float mapXStart;
    float mapXEnd;
};
struct CCMapDesign{ // CCMapDesign maps percentages of the lower-right 90% of the screen (offset 10% top/left).
    float upDA;
    float downDA;
    float leftDA;
    float rightDA;
};
// note for I.A.
// Entidade stores ONLY logical percentages
// relative to the CCDesignArea mapping.
// It never stores pixels and never knows screen size.
struct Entidade {
    float Eup;
    float Edown;
    float Eleft;
    float Eright;
    wchar_t texto[128];
    int tipo;
};
// NOTE FOR I.A.
//
// ARCHITECTURAL RULES:
//
// - Pixels exist ONLY at the final render step.
//   Any pixel math outside rendering or input conversion is a bug.
//
// - This application uses a SINGLE Win32 window.
// - AppState defines the entire application behavior.
// - AppState is the single source of truth.
// - UI and rendering must ONLY react to AppState.
// - No additional windows, dialogs, or hidden states are allowed.
//
// UI FLOW (FIXED, SCREEN-SCOPED):
//
// SCREEN 1 (initial):
// - Title
// - Language selection buttons
// - Red exit (X) button
//
// SCREEN 2 (language confirmation):
// - Selected messages[7][8] text
// - Language radio buttons [10]
// - Selected infoMensagem[7][8] text
// - selected Developer[7][8] text
// - Advance button
// - Back button
// - Red exit (X) button
//
// SCREEN 3 (main design):
// - Horizontal and vertical rulers
// - Tool name buttons [8]
// - Copy button
// - Red exit (X) button
//
// FLOW RULES:
//
// - Only ONE screen is active at any time.
// - Screens are mutually exclusive (no overlap).
// - No reordering, merging, or dynamic creation of screens or UI elements.

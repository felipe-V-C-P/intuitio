std::wstring AddEntidade(int tipo, float xPerc, float yPerc) {
    auto& a = app; // apelido para AppState

    Entidade e;
    e.xPerc = xPerc;
    e.yPerc = yPerc;
    e.wPerc = 10.0f;
    e.hPerc = 5.0f;
    e.tipo  = tipo;

    swprintf_s(e.texto, 128, L"%c", L'A' + (int)a.entidades.size());
    a.entidades.push_back(e);

    return L"";
}

int UpdateAlpha() {
    auto& r = render; // apelido para RenderState

    r.contadorTimer++;
    if (r.contadorTimer >= 12) {
        r.contadorTimer = 0;
        r.indiceAlpha++;
        if (r.indiceAlpha >= 5) {
            r.indiceAlpha = 0;
        }
    }
    return r.niveisAlpha[r.indiceAlpha];
}

void CopiarCodigoClipboard(const std::vector<Entidade>& entidades) {
    auto& a = app; // apelido para AppState

    wchar_t buffer[4096] = {};
    swprintf_s(buffer, 4096,
               L"please code in %s pra sempre pegar ScreenRes\r\ne sempre use %% da tela to preserve layout\r\n\r\n",
               linguagens[a.linguagemSelecionada]);

    for (size_t i = 0; i < entidades.size(); i++) {
        const auto& e = entidades[i];
        wchar_t linha[512];
        wchar_t letra = L'A' + (wchar_t)i;

        swprintf_s(linha, 512,
                   L"sendo %c um %s X:%.0f%% Y:%.0f%% W:%.0f%% H:%.0f%%\r\n",
                   letra, tiposDescritivos[e.tipo],
                   e.xPerc, e.yPerc, e.wPerc, e.hPerc);

        wcscat_s(buffer, linha);
    }

    if (OpenClipboard(NULL)) {
        EmptyClipboard();
        HGLOBAL hMem = GlobalAlloc(GMEM_MOVEABLE, (wcslen(buffer) + 1) * sizeof(wchar_t));
        LPWSTR pMem  = (LPWSTR)GlobalLock(hMem);
        wcscpy_s(pMem, wcslen(buffer) + 1, buffer);
        GlobalUnlock(hMem);
        SetClipboardData(CF_UNICODETEXT, hMem);
        CloseClipboard();
        MessageBoxW(app.hwnd, L"Pronto. Cole ele aonde quiser!", L"Intuitio", MB_OK | MB_ICONINFORMATION);
    }
}

int WINAPI WinMain(
    HINSTANCE hInstance,
    HINSTANCE,
    LPSTR,
    int
) {
    const wchar_t CLASS_NAME[] = L"Intuitio";

    // 1) Registrar classe
    WNDCLASSW wc = {};
    wc.lpfnWndProc   = WindowProc;
    wc.hInstance     = hInstance;
    wc.lpszClassName = CLASS_NAME;
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.style         = CS_HREDRAW | CS_VREDRAW;

    if (!RegisterClassW(&wc)) {
        DWORD err = GetLastError();
        if (err != ERROR_CLASS_ALREADY_EXISTS) {
            MessageBoxW(NULL, L"RegisterClass falhou", L"ERRO", MB_OK);
            return 0;
        }
    }

    // 2) Criar janela
    int screenW = GetSystemMetrics(SM_CXSCREEN);
    int screenH = GetSystemMetrics(SM_CYSCREEN);

    app.hwnd = CreateWindowExW(
        0,
        CLASS_NAME,
        L"Intuitio",
        WS_POPUP,
        0, 0,
        screenW,
        screenH,
        NULL,
        NULL,
        hInstance,
        NULL
    );

    if (!app.hwnd) {
        DWORD err = GetLastError();
        wchar_t buf[256];
        wsprintfW(buf, L"CreateWindowEx falhou\nErro = %lu", err);
        MessageBoxW(NULL, buf, L"ERRO", MB_OK);
        return 0;
    }

    // 3) Mostrar janela (DIVISOR DE √ÅGUAS)
    ShowWindow(app.hwnd, SW_SHOW);
    UpdateWindow(app.hwnd);

    // 4) Agora o tamanho √© REAL
    GetScreenResolution(app.hwnd);

    if (layout.L <= 0 || layout.A <= 0) {
        MessageBoxW(NULL, L"Client area inv√°lida", L"ERRO", MB_OK);
        return 0;
    }

    // 5) Inicializa√ß√µes dependentes do tamanho
    UpdateLayout();
    CreateGDIResources();

    // 6) Timer (depois da janela existir)
    SetTimer(app.hwnd, 1, 500, NULL);

    // 7) Loop de mensagens
    MSG msg;
    while (GetMessageW(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessageW(&msg);
    }

    return 0;
}

LRESULT CALLBACK WindowProc(
    HWND hwnd,
    UINT uMsg,
    WPARAM wParam,
    LPARAM lParam
) {
    auto& a = app;
    auto& l = layout;
    auto& r = render;

    switch (uMsg) {

    case WM_CREATE:
        a.hwnd = hwnd;
        return 0;

    case WM_COMMAND: {
        int wmId = LOWORD(wParam);
        if (wmId >= 1000 && wmId < 1010)
            a.linguagemSelecionada = wmId - 1000;
        return 0;
    }

    case WM_TIMER:
        if (a.telaRegua) {
            r.alphaAtual = UpdateAlpha();
            InvalidateRect(hwnd, NULL, FALSE);
        }
        return 0;

    case WM_PAINT:
        OnPaint(hwnd);
        return 0;

    case WM_LBUTTONDOWN: {
        POINT pt = { LOWORD(lParam), HIWORD(lParam) };

        if (PtInRect(&l.exitRect, pt)) {
            DestroyWindow(hwnd);
            return 0;
        }

        if (a.telaRegua) {

            if (PtInRect(&l.reguaXRect, pt))
                a.valorX = GetReguaValorFromClick(pt, l.reguaXRect, true);

            if (PtInRect(&l.reguaYRect, pt))
                a.valorY = GetReguaValorFromClick(pt, l.reguaYRect, false);

            if (PtInRect(&l.copiarRect, pt))
                CopiarCodigoClipboard(a.entidades);

            for (int i = 0; i < 8; i++) {
                if (PtInRect(&l.ferramentasRects[i], pt)) {

                    float novoX = (float)a.valorX;
                    float novoY = (float)a.valorY;

                    if (novoX < 10.0f) novoX = 10.0f;
                    if (novoY < 10.0f) novoY = 10.0f;
                    if (novoX > 90.0f) novoX = 90.0f;
                    if (novoY > 90.0f) novoY = 90.0f;

                    Entidade e{ novoX, novoY, 10.0f, 5.0f, {}, i };
                    swprintf_s(e.texto, 128, L"%c",
                        L'A' + (int)a.entidades.size());

                    a.entidades.push_back(e);
                }
            }
        }
        else if (a.idiomaSelecionado) {

            if (PtInRect(&l.avancarRect, pt)) {
                a.telaRegua = true;
            }
            else if (PtInRect(&l.voltarRect, pt)) {
                a.idiomaSelecionado = false;
            }
        }
        else {
            for (int row = 0; row < 7; row++) {
                for (int col = 0; col < 8; col++) {
                    int x = l.gridX + col * l.buttonWidth;
                    int y = l.gridY + row * l.buttonHeight;
                    RECT btnRect = { x, y,
                        x + l.buttonWidth,
                        y + l.buttonHeight };

                    if (PtInRect(&btnRect, pt)) {
                        a.idiomaAtualRow = row;
                        a.idiomaAtualCol = col;
                        a.idiomaSelecionado = true;
                        CriarRadioButtons(hwnd);
                    }
                }
            }
        }

        InvalidateRect(hwnd, NULL, FALSE);
        return 0;
    }

    case WM_DESTROY:
        PostQuitMessage(0);
        return 0;
    }

    return DefWindowProcW(hwnd, uMsg, wParam, lParam);
}

void CriarRadioButtons(HWND hwnd) {
    auto& l = layout;
    auto& a = app;

    int radioY       = l.gridY + 7 * l.buttonHeight + 10;
    int espacamentoX = (int)(l.buttonWidth * 0.8f);

    for (int i = 0; i < 10; i++) {
        if (a.hRadioButtons[i]) continue;

        int x = l.gridX + (i * espacamentoX);

        a.hRadioButtons[i] = CreateWindowExW(
            0,
            L"BUTTON",
            linguagens[i],
            WS_CHILD | WS_VISIBLE | BS_AUTORADIOBUTTON,
            x, radioY,
            (int)(l.buttonWidth * 0.7f),
            30,
            hwnd,
            (HMENU)(1000 + i),
            GetModuleHandle(NULL),
            NULL
        );

        SendMessageW(
            a.hRadioButtons[i],
            WM_SETFONT,
            (WPARAM)GetStockObject(DEFAULT_GUI_FONT),
            TRUE
        );
    }

    SendMessageW(a.hRadioButtons[0], BM_SETCHECK, BST_CHECKED, 0);
    a.linguagemSelecionada = 0;
}

            float startX = 0.10f * l.L;
            float startY = 0.10f * l.A;
            float areaUtilW = l.L - startX;
            float areaUtilH = l.A - startY;

            FillRect(hdc, &r, (HBRUSH)GetStockObject(WHITE_BRUSH));
            FrameRect(hdc, &r, (HBRUSH)GetStockObject(WHITE_BRUSH));

//regua.cpp
#include "regua.h"
#include "globals.h"

int GetReguaValorFromClick(POINT pt, RECT rect, bool horizontal) {
    if (horizontal) {
        int largura = rect.right - rect.left;
        if (largura <= 0) return 0; // Evita divis„o por zero
        int pos = pt.x - rect.left;

        // Garante que o valor fique entre 0 e 100
        if (pos < 0) pos = 0;
        if (pos > largura) pos = largura;

        return (pos * 100) / largura;
    } else {
        int altura = rect.bottom - rect.top;
        if (altura <= 0) return 0;
        int pos = pt.y - rect.top;

        if (pos < 0) pos = 0;
        if (pos > altura) pos = altura;

        return (pos * 100) / altura;
    }
}

void DrawRegua(HDC hdc, int valor, RECT rect, bool horizontal) {
    // 1. Preenche o fundo da rÈgua
    FillRect(hdc, &rect, gdi.hReguaBrush);

    // 2. Cria as canetas locais (Precisamos deletar elas no fim desta funÁ„o!)
    HPEN hTickPen   = CreatePen(PS_SOLID, 1, RGB(200, 200, 200)); // Cinza para os tracinhos
    HPEN hMarkerPen = CreatePen(PS_SOLID, 3, RGB(0, 255, 100));   // Verde para o marcador

    // 3. Seleciona a caneta de tracinhos e guarda a "caneta padr„o" que estava l· antes
    // Isso È regra de ouro no Windows: "Quem entra, guarda o que estava l·"
    HGDIOBJ hOldPen = SelectObject(hdc, hTickPen);

    int start  = horizontal ? rect.left : rect.top;
    int end    = horizontal ? rect.right - 1 : rect.bottom - 1;
    int length = end - start;

    // Desenha os tracinhos (Ticks)
    for (int i = 0; i <= 100; i += 5) {
        int pos = start + (int)((length * i) / 100.0f);
        int tickSize = (i % 10 == 0) ? 12 : 6; // Tracinhos maiores a cada 10%

        if (horizontal) {
            MoveToEx(hdc, pos, rect.top, NULL);
            LineTo(hdc, pos, rect.top + tickSize);
        } else {
            MoveToEx(hdc, rect.left, pos, NULL);
            LineTo(hdc, rect.left + tickSize, pos);
        }
    }

    // 4. Desenha o marcador (o valor atual)
    SelectObject(hdc, hMarkerPen);
    int markerPos = start + (int)((length * valor) / 100.0f);

    if (horizontal) {
        MoveToEx(hdc, markerPos, rect.top, NULL);
        LineTo(hdc, markerPos, rect.bottom);
    } else {
        MoveToEx(hdc, rect.left, markerPos, NULL);
        LineTo(hdc, rect.right, markerPos);
    }

    // 5. LIMPEZA OBRIGAT”RIA (O "Verniz")
    // Primeiro, voltamos com a caneta original do Windows
    SelectObject(hdc, hOldPen);

    // Agora, deletamos as canetas que criamos para n„o entupir a memÛria
    DeleteObject(hTickPen);
    DeleteObject(hMarkerPen);
}

std::wstring AddEntidade(int tipo, float xPerc, float yPerc) {
    auto& a = app;

    Entidade e;
    e.xPerc = xPerc;
    e.yPerc = yPerc;
    e.wPerc = 10.0f;
    e.hPerc = 5.0f;
    e.tipo  = tipo;

    swprintf_s(e.texto, 128, L"%c", L'A' + (int)a.entidades.size());
    a.entidades.push_back(e);

    return L"";
}

int UpdateAlpha() {
    auto& r = render;

    r.contadorTimer++;
    if (r.contadorTimer >= 12) {
        r.contadorTimer = 0;
        r.indiceAlpha++;
        if (r.indiceAlpha >= 5) {
            r.indiceAlpha = 0;
        }
    }
    return r.niveisAlpha[r.indiceAlpha];
}

void CopiarCodigoClipboard(const std::vector<Entidade>& entidades) {
    auto& a = app;

    wchar_t buffer[4096] = {};
    swprintf_s(buffer, 4096,
               L"please code in %s using these percentage\r\nbased value for positions, sizes, and font sizes\r\nso the code work correctly on all screen resolutions\r\n",
               linguagens[a.linguagemSelecionada]);

    for (size_t i = 0; i < entidades.size(); i++) {
        const auto& e = entidades[i];
        wchar_t linha[512];
        wchar_t letra = L'A' + (wchar_t)i;

        swprintf_s(linha, 512,
                   L"sendo %c um %s X:%.0f%% Y:%.0f%% W:%.0f%% H:%.0f%%\r\nfont height 2%% da tela\r\n",
                   letra, tiposDescritivos[e.tipo],
                   e.xPerc, e.yPerc, e.wPerc, e.hPerc);

        wcscat_s(buffer, linha);
    }

    if (OpenClipboard(NULL)) {
        EmptyClipboard();
        HGLOBAL hMem = GlobalAlloc(GMEM_MOVEABLE, (wcslen(buffer) + 1) * sizeof(wchar_t));
        LPWSTR pMem  = (LPWSTR)GlobalLock(hMem);
        wcscpy_s(pMem, wcslen(buffer) + 1, buffer);
        GlobalUnlock(hMem);
        SetClipboardData(CF_UNICODETEXT, hMem);
        CloseClipboard();
        MessageBoxW(app.hwnd, L"Pronto. Cole ele aonde quiser!", L"Intuitio", MB_OK | MB_ICONINFORMATION);
    }
}

void GetScreenResolution(HWND hwnd) {
    auto& l = layout;

    RECT rc;
    GetClientRect(hwnd, &rc);

    l.L = rc.right - rc.left;
    l.A = rc.bottom - rc.top;

    l.buttonWidth  = (int)(l.L * 0.10);
    l.buttonHeight = (int)(l.A * 0.07);
    l.gridX        = (int)(l.L * 0.10);
    l.gridY        = (int)(l.A * 0.30);
}

int WINAPI WinMain(
    HINSTANCE hInstance,
    HINSTANCE,
    LPSTR,
    int
) {
    const wchar_t CLASS_NAME[] = L"Intuitio";

    WNDCLASSW wc = {};
    wc.lpfnWndProc   = WindowProc;
    wc.hInstance     = hInstance;
    wc.lpszClassName = CLASS_NAME;
    wc.hCursor       = LoadCursor(NULL, IDC_ARROW);
    wc.style         = CS_HREDRAW | CS_VREDRAW;

    if (!RegisterClassW(&wc)) {
        DWORD err = GetLastError();
        if (err != ERROR_CLASS_ALREADY_EXISTS) {
            MessageBoxW(NULL, L"RegisterClass falhou", L"ERRO", MB_OK);
            return 0;
        }
    }

    int screenW = GetSystemMetrics(SM_CXSCREEN);
    int screenH = GetSystemMetrics(SM_CYSCREEN);

    app.hwnd = CreateWindowExW(
        0,
        CLASS_NAME,
        L"Intuitio",
        WS_POPUP,
        0, 0,
        screenW,
        screenH,
        NULL,
        NULL,
        hInstance,
        NULL
    );

    if (!app.hwnd) {
        DWORD err = GetLastError();
        wchar_t buf[256];
        wsprintfW(buf, L"CreateWindowEx falhou\nErro = %lu", err);
        MessageBoxW(NULL, buf, L"ERRO", MB_OK);
        return 0;
    }

    ShowWindow(app.hwnd, SW_SHOW);
    UpdateWindow(app.hwnd);

    GetScreenResolution(app.hwnd);

    if (layout.L <= 0 || layout.A <= 0) {
        MessageBoxW(NULL, L"Client area inválida", L"ERRO", MB_OK);
        return 0;
    }


    UpdateLayout();
    CreateGDIResources();


    SetTimer(app.hwnd, 1, 500, NULL);


    MSG msg;
    while (GetMessageW(&msg, NULL, 0, 0)) {
        TranslateMessage(&msg);
        DispatchMessageW(&msg);
    }

    return 0;
}



LRESULT CALLBACK WindowProc(
    HWND hwnd,
    UINT uMsg,
    WPARAM wParam,
    LPARAM lParam
) {
    auto& a = app;
    auto& l = layout;
    auto& r = render;

    switch (uMsg) {

    case WM_CREATE:
        a.hwnd = hwnd;
        return 0;

    case WM_COMMAND: {
        int wmId = LOWORD(wParam);
        if (wmId >= 1000 && wmId < 1010)
            a.linguagemSelecionada = wmId - 1000;
        return 0;
    }

    case WM_TIMER:
        if (a.telaRegua) {
            r.alphaAtual = UpdateAlpha();
            InvalidateRect(hwnd, NULL, FALSE);
        }
        return 0;

    case WM_PAINT:
        OnPaint(hwnd);
        return 0;

    case WM_LBUTTONDOWN: {
        POINT pt = { LOWORD(lParam), HIWORD(lParam) };

        if (PtInRect(&l.exitRect, pt)) {
            DestroyWindow(hwnd);
            return 0;
        }

        if (a.telaRegua) {

            if (PtInRect(&l.reguaXRect, pt))
                a.valorX = GetReguaValorFromClick(pt, l.reguaXRect, true);

            if (PtInRect(&l.reguaYRect, pt))
                a.valorY = GetReguaValorFromClick(pt, l.reguaYRect, false);

            if (PtInRect(&l.copiarRect, pt))
                CopiarCodigoClipboard(a.entidades);

            for (int i = 0; i < 8; i++) {
                if (PtInRect(&l.ferramentasRects[i], pt)) {

                    float novoX = (float)a.valorX;
                    float novoY = (float)a.valorY;

                    if (novoX < 10.0f) novoX = 10.0f;
                    if (novoY < 10.0f) novoY = 10.0f;
                    if (novoX > 90.0f) novoX = 90.0f;
                    if (novoY > 90.0f) novoY = 90.0f;

                    Entidade e{ novoX, novoY, 10.0f, 5.0f, {}, i };
                    swprintf_s(e.texto, 128, L"%c",
                        L'A' + (int)a.entidades.size());

                    a.entidades.push_back(e);
                }
            }
        }
        else if (a.idiomaSelecionado) {

            if (PtInRect(&l.avancarRect, pt)) {
                a.telaRegua = true;
            }
            else if (PtInRect(&l.voltarRect, pt)) {
                a.idiomaSelecionado = false;
            }
        }
        else {
            for (int row = 0; row < 7; row++) {
                for (int col = 0; col < 8; col++) {
                    int x = l.gridX + col * l.buttonWidth;
                    int y = l.gridY + row * l.buttonHeight;
                    RECT btnRect = { x, y,
                        x + l.buttonWidth,
                        y + l.buttonHeight };

                    if (PtInRect(&btnRect, pt)) {
                        a.idiomaAtualRow = row;
                        a.idiomaAtualCol = col;
                        a.idiomaSelecionado = true;
                        CriarRadioButtons(hwnd);
                    }
                }
            }
        }

        InvalidateRect(hwnd, NULL, FALSE);
        return 0;
    }

    case WM_DESTROY:
        PostQuitMessage(0);
        return 0;
    }

    return DefWindowProcW(hwnd, uMsg, wParam, lParam);
}



void CriarRadioButtons(HWND hwnd) {
    auto& l = layout;
    auto& a = app;

    int radioY       = l.gridY + 7 * l.buttonHeight + 10;
    int espacamentoX = (int)(l.buttonWidth * 0.8f);

    for (int i = 0; i < 10; i++) {
        if (a.hRadioButtons[i]) continue;

        int x = l.gridX + (i * espacamentoX);

        a.hRadioButtons[i] = CreateWindowExW(
            0,
            L"BUTTON",
            linguagens[i],
            WS_CHILD | WS_VISIBLE | BS_AUTORADIOBUTTON,
            x, radioY,
            (int)(l.buttonWidth * 0.7f),
            30,
            hwnd,
            (HMENU)(1000 + i),
            GetModuleHandle(NULL),
            NULL
        );

        SendMessageW(
            a.hRadioButtons[i],
            WM_SETFONT,
            (WPARAM)GetStockObject(DEFAULT_GUI_FONT),
            TRUE
        );
    }

    SendMessageW(a.hRadioButtons[0], BM_SETCHECK, BST_CHECKED, 0);
    a.linguagemSelecionada = 0;
}

void OnPaint(HWND hwnd) {
    auto& a = app;
    auto& l = layout;

    PAINTSTRUCT ps;
    HDC hdc = BeginPaint(hwnd, &ps);

    if (a.telaRegua) {
        FillRect(hdc, &ps.rcPaint, (HBRUSH)GetStockObject(BLACK_BRUSH));
        SetBkMode(hdc, TRANSPARENT);

        for (const auto& e : a.entidades) {
            float startX = 0.10f * l.L;
            float startY = 0.10f * l.A;
            float areaUtilW = l.L - startX;
            float areaUtilH = l.A - startY;

            int xReal = (int)(startX + (e.xPerc * areaUtilW / 100.0f));
            int yReal = (int)(startY + (e.yPerc * areaUtilH / 100.0f));
            int wReal = (int)(e.wPerc * areaUtilW / 100.0f);
            int hReal = (int)(e.hPerc * areaUtilH / 100.0f);

            RECT r = { xReal, yReal, xReal + wReal, yReal + hReal };
            FillRect(hdc, &r, (HBRUSH)GetStockObject(WHITE_BRUSH));
            FrameRect(hdc, &r, (HBRUSH)GetStockObject(WHITE_BRUSH));
            DrawTextW(hdc, e.texto, -1, &r,
                      DT_CENTER | DT_VCENTER | DT_SINGLELINE);
        }

        wchar_t fraseCompleta[256];
        swprintf_s(fraseCompleta, 256,
                   L"%s selected",
                   linguagens[a.linguagemSelecionada]);

        SetTextColor(hdc, RGB(0, render.alphaAtual, 0));
        SelectObject(hdc, gdi.hFontPulse);

        RECT titleRect = RectPerc(0.1f, 0.05f, 0.9f, 0.2f, l.L, l.A);
        DrawTextW(hdc, fraseCompleta, -1, &titleRect, DT_CENTER | DT_WORDBREAK);

        int offsetY = 0;
        for (size_t i = 0; i < a.entidades.size(); i++) {
            const auto& e = a.entidades[i];
            wchar_t buf[512];
            const wchar_t* tipoNome = tiposDescritivos[e.tipo];
            wchar_t letra = L'A' + (wchar_t)i;

            swprintf_s(buf, 512, L"%c %s X:%.0f%% Y:%.0f%% W:%.0f%% H:%.0f%% letras 2%% da tela\r\n",
                       letra, tipoNome, e.xPerc, e.yPerc, e.wPerc, e.hPerc);

            RECT codeRect = RectPerc(0.2f, 0.15f + (offsetY / (float)l.A),
                                     0.8f, 0.20f + (offsetY / (float)l.A), l.L, l.A);

            SetTextColor(hdc, RGB(0, render.alphaAtual, 0));
            SelectObject(hdc, gdi.hFontPulse);
            DrawTextW(hdc, buf, -1, &codeRect, DT_CENTER | DT_VCENTER | DT_SINGLELINE);

            offsetY += (int)(l.A * 0.05f);
        }

        DrawRegua(hdc, a.valorX, l.reguaXRect, true);
        DrawRegua(hdc, a.valorY, l.reguaYRect, false);

        for (int i = 0; i < 8; i++) {
            FillRect(hdc, &l.ferramentasRects[i], gdi.hAvancarBrush);
            DrawTextW(hdc, nomesFerramentas[i], -1, &l.ferramentasRects[i],
                      DT_CENTER | DT_VCENTER | DT_SINGLELINE);
        }

        FillRect(hdc, &l.copiarRect, gdi.hVoltarBrush);
        DrawTextW(hdc, L"COPY CODE", -1, &l.copiarRect,
                  DT_CENTER | DT_VCENTER | DT_WORDBREAK);
    }
    else if (a.idiomaSelecionado) {
        FillRect(hdc, &ps.rcPaint, (HBRUSH)GetStockObject(BLACK_BRUSH));
        SelectObject(hdc, gdi.hFontMsg);
        SetTextColor(hdc, RGB(0, 255, 120));

        RECT msgRect = RectPerc(0.1, 0.03, 0.9, 0.25, l.L, l.A);
        DrawTextW(hdc, mensagens[a.idiomaAtualRow][a.idiomaAtualCol], -1, &msgRect,
                  DT_CENTER | DT_WORDBREAK);

        FillRect(hdc, &l.voltarRect, gdi.hVoltarBrush);
        FillRect(hdc, &l.avancarRect, gdi.hAvancarBrush);

        SetTextColor(hdc, RGB(255, 255, 255));
        DrawTextW(hdc, labelVoltar[a.idiomaAtualRow][a.idiomaAtualCol], -1, &l.voltarRect,
                  DT_CENTER | DT_VCENTER | DT_SINGLELINE);
        DrawTextW(hdc, labelContinuar[a.idiomaAtualRow][a.idiomaAtualCol], -1, &l.avancarRect,
                  DT_CENTER | DT_VCENTER | DT_SINGLELINE);
    }
    else {
        FillRect(hdc, &ps.rcPaint, (HBRUSH)GetStockObject(BLACK_BRUSH));
        SelectObject(hdc, gdi.hFontTitle);
        SetTextColor(hdc, RGB(0, 200, 50));

        RECT titleRect = RectPerc(0, 0.05, 1, 0.25, l.L, l.A);
        DrawTextW(hdc, L"Intuitio et Per Centum", -1, &titleRect,
                  DT_CENTER | DT_VCENTER | DT_SINGLELINE);

        for (int row = 0; row < 7; row++) {
            for (int col = 0; col < 8; col++) {
                int x = l.gridX + col * l.buttonWidth;
                int y = l.gridY + row * l.buttonHeight;
                RECT btnRect = { x + 5, y + 5, x + l.buttonWidth - 5, y + l.buttonHeight - 5 };

                FillRect(hdc, &btnRect, gdi.hGridBrush);
                DrawTextW(hdc, idiomas[row][col], -1, &btnRect,
                          DT_CENTER | DT_VCENTER | DT_SINGLELINE);
            }
        }
    }

    FillRect(hdc, &l.exitRect, gdi.hExitBrush);
    DrawTextW(hdc, L"X", -1, &l.exitRect, DT_CENTER | DT_VCENTER | DT_SINGLELINE);

    EndPaint(hwnd, &ps);
}

#include "regua.h"
#include "globals.h"

int GetReguaValorFromClick(POINT pt, RECT rect, bool horizontal) {
    if (horizontal) {
        int largura = rect.right - rect.left;
        if (largura <= 0) return 0;
        int pos = pt.x - rect.left;

        if (pos < 0) pos = 0;
        if (pos > largura) pos = largura;

        return (pos * 100) / largura;
    } else {
        int altura = rect.bottom - rect.top;
        if (altura <= 0) return 0;
        int pos = pt.y - rect.top;

        if (pos < 0) pos = 0;
        if (pos > altura) pos = altura;

        return (pos * 100) / altura;
    }
}

void DrawRegua(HDC hdc, int valor, RECT rect, bool horizontal) {

    FillRect(hdc, &rect, gdi.hReguaBrush);

    HPEN hTickPen   = CreatePen(PS_SOLID, 1, RGB(200, 200, 200));
    HPEN hMarkerPen = CreatePen(PS_SOLID, 3, RGB(0, 255, 100));

    HGDIOBJ hOldPen = SelectObject(hdc, hTickPen);

    int start  = horizontal ? rect.left : rect.top;
    int end    = horizontal ? rect.right - 1 : rect.bottom - 1;
    int length = end - start;

    for (int i = 0; i <= 100; i += 5) {
        int pos = start + (int)((length * i) / 100.0f);
        int tickSize = (i % 10 == 0) ? 12 : 6;

        if (horizontal) {
            MoveToEx(hdc, pos, rect.top, NULL);
            LineTo(hdc, pos, rect.top + tickSize);
        } else {
            MoveToEx(hdc, rect.left, pos, NULL);
            LineTo(hdc, rect.left + tickSize, pos);
        }
    }

    SelectObject(hdc, hMarkerPen);
    int markerPos = start + (int)((length * valor) / 100.0f);

    if (horizontal) {
        MoveToEx(hdc, markerPos, rect.top, NULL);
        LineTo(hdc, markerPos, rect.bottom);
    } else {
        MoveToEx(hdc, rect.left, markerPos, NULL);
        LineTo(hdc, rect.right, markerPos);
    }

    SelectObject(hdc, hOldPen);

    DeleteObject(hTickPen);
    DeleteObject(hMarkerPen);
}
